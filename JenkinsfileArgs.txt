pipeline {
    agent any
    parameters {
        string(name: 'NUM1', defaultValue: '5', description: 'First number')
        string(name: 'NUM2', defaultValue: '10', description: 'Second number')
    }
    stages {
        stage('Compile') {
            steps {
                bat 'javac Addition.java'
            }
        }
        stage('Run') {
            steps {
                bat "java Addition ${params.NUM1} ${params.NUM2}"
            }
        }
    }
}
